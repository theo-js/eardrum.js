API

Create a new butterfly and use it in eardrum.configure method

Code:

var myObject = {};
var newButterfly = createButterfly();

eardrum.configure({
  object: myObject,
  property: 'property',
  handler: function (mouseEvent, config) {
    moveButterfly(mouseEvent, newButterfly);
  },
  listener: {
    target: window,
  type: 'mousemove'
  },
  additionalRefProperties: {
    butterfly: newButterfly
  }
});
